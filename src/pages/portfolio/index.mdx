import Layout from "../../components/Layout"
import { graphql, Link } from "gatsby"

export default ({ children, ...props }) => {
  const { nodes } = props.pageResources.json.data.allMdx
  return (
    <Layout>
      {children}
      {nodes.map(({ slug }) => (
        <div key={slug}>
          <a as={Link} to={"/" + slug}>
            {slug}
          </a>
        </div>
      ))}
    </Layout>
  )
}

export const pageQuery = graphql`
  query PortfolioQuery {
    allMdx(filter: { slug: { regex: "/^portfolio/./" } }) {
      nodes {
        slug
      }
    }
  }
`
